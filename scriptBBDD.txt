
DROP DATABASE DatingApp;
CREATE DATABASE DatingApp;
USE DatingApp;


-- Tabla User
CREATE TABLE User (
    IdUser INT AUTO_INCREMENT PRIMARY KEY,
    Nombre VARCHAR(50) NOT NULL,
    Apellido1 VARCHAR(50) NOT NULL,
    Apellido2 VARCHAR(50) NOT NULL,
    Alias VARCHAR(50) NOT NULL,
    Pass VARCHAR(255) NOT NULL, -- Contraseña encriptada
    Naixement DATE NOT NULL,
    Orientacio ENUM('Heterosexual', 'Homosexual', 'Bisexual') NOT NULL, -- Preferencia sexual
    Sexe ENUM('Hombre', 'Mujer', 'No binario') NOT NULL, -- Género
    Longitud VARCHAR(50) NOT NULL,
    Latitud VARCHAR(50) NOT NULL,
    Email VARCHAR(100) UNIQUE NOT NULL,
    EdadUser INT NOT NULL,
    EdadMax INT NOT NULL,
    EdadMin INT NOT NULL,
    ModifyAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP NOT NULL,
    CreateAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL
);
-- Tabla Photo
CREATE TABLE Photo (
    idFoto INT AUTO_INCREMENT PRIMARY KEY,
    idUser INT NOT NULL,
    URL VARCHAR(255) NOT NULL,
    FOREIGN KEY (idUser) REFERENCES User(IdUser) ON DELETE CASCADE
);

-- Tabla Like
CREATE TABLE `Like` (
    LikeId INT AUTO_INCREMENT PRIMARY KEY,
    idUser INT NOT NULL,
    idUserLiked INT NOT NULL,
    CreateAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (idUser) REFERENCES User(IdUser) ON DELETE CASCADE,
    FOREIGN KEY (idUserLiked) REFERENCES User(IdUser) ON DELETE CASCADE
);

-- Tabla Match
CREATE TABLE Matches (
    idMatch INT AUTO_INCREMENT PRIMARY KEY,
    isUser1 INT NOT NULL,
    idUser2 INT NOT NULL,
    MatchAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (isUser1) REFERENCES User(IdUser) ON DELETE CASCADE,
    FOREIGN KEY (idUser2) REFERENCES User(IdUser) ON DELETE CASCADE
);

-- Tabla Message
CREATE TABLE Message (
    idMessage INT AUTO_INCREMENT PRIMARY KEY,
    MatchId INT NOT NULL,
    isUserReceived INT NOT NULL,
    idUserSend INT NOT NULL,
    text TEXT NOT NULL,
    SentAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (MatchId) REFERENCES Matches(idMatch) ON DELETE CASCADE,
    FOREIGN KEY (isUserReceived) REFERENCES User(IdUser) ON DELETE CASCADE,
    FOREIGN KEY (idUserSend) REFERENCES User(IdUser) ON DELETE CASCADE
);

-- Crear Evento Para Actualizacion de la edad.
SET GLOBAL event_scheduler = ON;

CREATE EVENT actualizar_edades_diarias
ON SCHEDULE EVERY 1 DAY
STARTS CURRENT_TIMESTAMP  -- Esto establecerá el inicio del evento a la hora actual
DO
  UPDATE User
  SET Edad = TIMESTAMPDIFF(YEAR, Naixement, CURDATE());
