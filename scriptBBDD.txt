DROP DATABASE DatingApp;
CREATE DATABASE DatingApp;
USE DatingApp;

-- User Table
CREATE TABLE User (
    IdUser INT AUTO_INCREMENT PRIMARY KEY,
    FirstName VARCHAR(50) NOT NULL,
    LastName1 VARCHAR(50) NOT NULL,
    LastName2 VARCHAR(50) NOT NULL,
    Username VARCHAR(50) NOT NULL,
    Password VARCHAR(255) NOT NULL, -- Encrypted password
    BirthDate DATE NOT NULL,
    Orientation ENUM('Heterosexual', 'Homosexual', 'Bisexual') NOT NULL, -- Sexual orientation
    Gender ENUM('Hombre', 'Mujer', 'No Binario') NOT NULL, -- Gender identity
    Longitude VARCHAR(50) NOT NULL,
    Latitude VARCHAR(50) NOT NULL,
    Email VARCHAR(100) UNIQUE NOT NULL,
    UserAge INT NOT NULL,
    MaxAge INT NOT NULL,
    MinAge INT NOT NULL,
    Points INT NOT NULL,
    Bio VARCHAR(100),
    Role ENUM('Admin','User') NOT NULL,
    ModifyAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP NOT NULL,
    CreateAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL
);

-- Photo Table
CREATE TABLE Photo (
    PhotoId INT AUTO_INCREMENT PRIMARY KEY,
    UserId INT NOT NULL,
    URL VARCHAR(255) NOT NULL,
    FOREIGN KEY (UserId) REFERENCES User(IdUser) ON DELETE CASCADE
);

-- Like Table
CREATE TABLE UserLikes (
    LikeId INT AUTO_INCREMENT PRIMARY KEY,
    UserId INT NOT NULL,
    LikedUserId INT NOT NULL,
    CreateAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (UserId) REFERENCES User(IdUser) ON DELETE CASCADE,
    FOREIGN KEY (LikedUserId) REFERENCES User(IdUser) ON DELETE CASCADE
);

-- Match Table
CREATE TABLE Matches (
    MatchId INT AUTO_INCREMENT PRIMARY KEY,
    User1Id INT NOT NULL,
    User2Id INT NOT NULL,
    MatchAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (User1Id) REFERENCES User(IdUser) ON DELETE CASCADE,
    FOREIGN KEY (User2Id) REFERENCES User(IdUser) ON DELETE CASCADE
);

-- Message Table
CREATE TABLE Message (
    MessageId INT AUTO_INCREMENT PRIMARY KEY,
    MatchId INT NOT NULL,
    ReceiverUserId INT NOT NULL,
    SenderUserId INT NOT NULL,
    Text TEXT NOT NULL,
    SentAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (MatchId) REFERENCES Matches(MatchId) ON DELETE CASCADE,
    FOREIGN KEY (ReceiverUserId) REFERENCES User(IdUser) ON DELETE CASCADE,
    FOREIGN KEY (SenderUserId) REFERENCES User(IdUser) ON DELETE CASCADE
);



SET GLOBAL event_scheduler = ON;

CREATE EVENT daily_age_update
ON SCHEDULE EVERY 1 DAY
STARTS CURRENT_TIMESTAMP  -- This will set the event start time to the current time
DO
  UPDATE User
  SET UserAge = TIMESTAMPDIFF(YEAR, BirthDate, CURDATE());
